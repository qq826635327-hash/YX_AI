```python
# tests/unit/test_business_service.py
from app.services.business_service import (
    list_by_project, get_one, create_entity, update_entity, delete_entity
)
from app.models.character import Character

class TestBusinessService:
    """business_service 核心 CRUD 单元测试"""

    def test_tc_unit_biz_001_list(self, session, project, character):
        """TC-UNIT-BIZ-001: list_by_project 正常返回"""
        results = list_by_project(session, Character, project.id)
        assert len(results) >= 1
        assert results[0].project_id == project.id

    def test_tc_unit_biz_002_list_empty(self, session, project):
        """TC-UNIT-BIZ-002: list_by_project 空结果"""
        results = list_by_project(session, Character, project.id)
        assert results == []

    def test_tc_unit_biz_003_create(self, session, project):
        """TC-UNIT-BIZ-003: create_entity 正常创建"""
        entity = create_entity(
            session, Character, project.id,
            {"name": "新角色", "char_type": "protagonist"}
        )
        assert entity.name == "新角色"
        assert entity.project_id == project.id

    def test_tc_unit_biz_004_create_duplicate(self, session, project, character):
        """TC-UNIT-BIZ-004: create_entity 重名冲突"""
        from sqlalchemy.exc import IntegrityError
        with pytest.raises((IntegrityError, ValueError)):
            create_entity(
                session, Character, project.id,
                {"name": character.name, "char_type": "protagonist"}
            )

    def test_tc_unit_biz_005_update(self, session, project, character):
        """TC-UNIT-BIZ-005: update_entity 正常更新"""
        updated = update_entity(
            session, Character, character.id,
            {"name": "更新后名称"}
        )
        assert updated is not None
        assert updated.name == "更新后名称"

    def test_tc_unit_biz_006_update_not_found(self, session, project):
        """TC-UNIT-BIZ-006: update_entity 不存在"""
        result = update_entity(session, Character, 99999, {"name": "X"})
        assert result is None

    def test_tc_unit_biz_008_delete(self, session, project, character):
        """TC-UNIT-BIZ-008: delete_entity 正常删除"""
        result = delete_entity(session, Character, character.id)
        assert result is True
        # 验证已删除
        assert get_one(session, Character, character.id) is None

    def test_tc_unit_biz_009_delete_not_found(self, session, project):
        """TC-UNIT-BIZ-009: delete_entity 不存在"""
        result = delete_entity(session, Character, 99999)
        assert result is False

    def test_tc_unit_biz_010_get_one(self, session, project, character):
        """TC-UNIT-BIZ-010: get_one 正常返回"""
        entity = get_one(session, Character, character.id)
        assert entity is not None
        assert entity.id == character.id

    def test_tc_unit_biz_011_get_one_not_found(self, session, project):
        """TC-UNIT-BIZ-011: get_one 不存在"""
        entity = get_one(session, Character, 99999)
        assert entity is None
```

### 3.16 Provider注册与Handler测试

| 编号 | 名称 | 目标 | 步骤 | 预期结果 | 优先级 |
|------|------|------|------|----------|--------|
| TC-UNIT-PROV-001 | 注册 Handler | providers/__init__.py 注册 | 导入 providers 模块 | agnes_handler 已注册 | P0 |
| TC-UNIT-PROV-002 | list_registered | providers.registry.list_registered | 调用 list_registered() | 返回已注册 kind 列表 | P0 |
| TC-UNIT-PROV-003 | get_handler_class 已注册 | providers.registry.get_handler_class | 传入 "agnes" | 返回 AgnesHandler 类 | P0 |
| TC-UNIT-PROV-004 | get_handler_class 未注册 | 同上 | 传入 "unknown" | 返回 None | P0 |
| TC-UNIT-PROV-005 | Handler.get_capabilities | AgnesHandler.get_capabilities() | 调用 get_capabilities() | 返回 ProviderCapabilitiesView | P0 |
| TC-UNIT-PROV-006 | Handler.validate_params | AgnesHandler.validate_params | 传入无效 model | 返回错误列表 | P1 |
| TC-UNIT-PROV-007 | openai 别名映射 | providers.registry.get_handler_class | 传入 "openai" | 返回 AgnesHandler（别名） | P1 |

```python
# tests/unit/test_provider_registry.py
from app.providers.registry import list_registered, get_handler_class

class TestProviderRegistry:
    """Provider 注册与 Handler 单元测试"""

    def test_tc_unit_prov_001_handler_registered(self):
        """TC-UNIT-PROV-001: 注册 Handler"""
        import app.providers  # noqa: F401 触发注册
        registered = list_registered()
        assert "agnes" in registered

    def test_tc_unit_prov_002_list_registered(self):
        """TC-UNIT-PROV-002: list_registered 返回已注册列表"""
        result = list_registered()
        assert isinstance(result, list)
        assert len(result) >= 1

    def test_tc_unit_prov_003_get_handler(self):
        """TC-UNIT-PROV-003: get_handler_class 已注册"""
        handler_cls = get_handler_class("agnes")
        assert handler_cls is not None
        assert handler_cls.__name__ == "AgnesHandler"

    def test_tc_unit_prov_004_get_handler_unknown(self):
        """TC-UNIT-PROV-004: get_handler_class 未注册"""
        handler_cls = get_handler_class("unknown")
        assert handler_cls is None

    def test_tc_unit_prov_005_capabilities(self):
        """TC-UNIT-PROV-005: Handler.get_capabilities"""
        handler_cls = get_handler_class("agnes")
        caps = handler_cls.get_capabilities()
        assert caps is not None
        # 验证返回 ProviderCapabilitiesView 结构
        assert hasattr(caps, "supports_first_frame") or hasattr(caps, "__dict__")

    def test_tc_unit_prov_006_validate_params(self):
        """TC-UNIT-PROV-006: Handler.validate_params 无效参数"""
        handler_cls = get_handler_class("agnes")
        errors = handler_cls.validate_params({"model": ""})
        assert isinstance(errors, list)
        assert len(errors) >= 1

    def test_tc_unit_prov_007_openai_alias(self):
        """TC-UNIT-PROV-007: openai 别名映射"""
        handler_cls = get_handler_class("openai")
        agnes_cls = get_handler_class("agnes")
        assert handler_cls is agnes_cls
```

```python
# tests/unit/test_script_parser.py
from app.pipelines.script_parser import parse_script

class TestScriptParser:
    """剧本解析管线单元测试"""

    def test_tc_unit_pipe_001_standard_format(self):
        """TC-UNIT-PIPE-001: 规则解析-标准格式"""
        script_text = (
            "第一集\n"
            "场景：城市街道\n"
            "角色A：你好\n"
            "角色B：你好，再见\n"
        )
        result = parse_script(script_text)
        assert "episodes" in result or "scenes" in result
        assert len(result.get("episodes", [])) >= 1

    def test_tc_unit_pipe_002_empty_text(self):
        """TC-UNIT-PIPE-002: 规则解析-空文本"""
        result = parse_script("")
        assert result == {} or result.get("episodes") == []

    def test_tc_unit_pipe_003_non_standard_format(self):
        """TC-UNIT-PIPE-003: 规则解析-非标准格式"""
        script_text = "这是一段非标准格式的文本内容，没有明确的场景和角色标记。"
        result = parse_script(script_text)
        # 不应崩溃，尽力返回结果
        assert isinstance(result, dict)

    def test_tc_unit_pipe_004_result_structure(self):
        """TC-UNIT-PIPE-004: 解析结果结构验证"""
        script_text = (
            "第一集 命运的开始\n"
            "场景：雨夜小巷\n"
            "林小雨：这是哪里？\n"
            "张明远：别怕，跟我来。\n"
        )
        result = parse_script(script_text)
        # 验证 ParsedResult 结构
        assert isinstance(result, dict)
        if "episodes" in result:
            for ep in result["episodes"]:
                assert "episode_no" in ep or "title" in ep
```

### 3.18 性能与压力测试

| 编号 | 名称 | 目标 | 步骤 | 预期结果 | 优先级 |
|------|------|------|------|----------|--------|
| TC-PERF-001 | 项目列表 100 条 | GET /api/projects | 预置 100 个项目，请求列表 | 响应时间 < 500ms | P1 |
| TC-PERF-002 | 角色列表 500 条 | GET /api/projects/{id}/characters | 预置 500 个角色，请求列表 | 响应时间 < 1s | P1 |
| TC-PERF-003 | 批量生成 50 任务 | POST /api/generate/batch | 提交 50 个生成任务 | 全部创建成功，无超时 | P2 |
| TC-PERF-004 | WebSocket 50 客户端 | 50个并发WS连接 | 同时建立 50 个 WS 连接 | 所有连接稳定，消息延迟 < 200ms | P2 |
| TC-PERF-005 | 日志查询 10000 条 | GET /api/logs?limit=200 | 预置 10000 条日志 | 响应时间 < 500ms | P2 |
| TC-PERF-006 | 数据库并发写入 | 10 线程同时创建实体 | 10 个并发请求创建角色 | 无死锁，全部成功 | P2 |

```python
# tests/perf/test_performance.py
import asyncio
import time
import pytest
from httpx import AsyncClient

class TestPerformance:
    """性能与压力测试"""

    @pytest.mark.asyncio
    async def test_tc_perf_001_project_list_100(
        self, client: AsyncClient, session, project_factory
    ):
        """TC-PERF-001: 项目列表 100 条性能测试"""
        project_factory.create_batch(100)
        start = time.perf_counter()
        response = await client.get("/api/projects")
        elapsed = time.perf_counter() - start
        assert response.status_code == 200
        assert elapsed < 0.5, f"响应时间 {elapsed:.3f}s 超过 500ms 限制"

    @pytest.mark.asyncio
    async def test_tc_perf_002_character_list_500(
        self, client: AsyncClient, session, project
    ):
        """TC-PERF-002: 角色列表 500 条性能测试"""
        from tests.factories import CharacterFactory
        for i in range(500):
            char = CharacterFactory(project_id=project.id, name=f"角色_{i}")
            session.add(char)
        session.commit()
        start = time.perf_counter()
        response = await client.get(
            f"/api/projects/{project.id}/characters"
        )
        elapsed = time.perf_counter() - start
        assert response.status_code == 200
        assert elapsed < 1.0, f"响应时间 {elapsed:.3f}s 超过 1s 限制"

    @pytest.mark.asyncio
    async def test_tc_perf_003_batch_generate_50(
        self, client: AsyncClient, project, mock_provider, session, episode
    ):
        """TC-PERF-003: 批量生成 50 任务"""
        from tests.factories import ShotFactory
        shots = []
        for i in range(50):
            shot = ShotFactory(
                project_id=project.id,
                episode_id=episode.id,
                shot_no=i + 1,
                first_frame_prompt=f"Test prompt {i}"
            )
            session.add(shot)
            shots.append(shot)
        session.commit()

        tasks = [
            {
                "target_type": "first_frame",
                "target_id": shot.id,
                "provider_id": mock_provider.id
            }
            for shot in shots
        ]
        start = time.perf_counter()
        response = await client.post("/api/generate/batch", json={"tasks": tasks})
        elapsed = time.perf_counter() - start
        assert response.status_code == 200
        assert elapsed < 10.0, f"批量创建耗时 {elapsed:.3f}s"

    @pytest.mark.asyncio
    async def test_tc_perf_004_websocket_50_clients(self):
        """TC-PERF-004: WebSocket 50 客户端并发连接"""
        import websockets
        clients = await asyncio.gather(*[
            websockets.connect("ws://localhost:8000/ws/tasks")
            for _ in range(50)
        ])
        try:
            assert len(clients) == 50
            # 向所有客户端发送消息测试延迟
            start = time.perf_counter()
            messages = await asyncio.gather(*[
                asyncio.wait_for(ws.recv(), timeout=5.0)
                for ws in clients
            ], return_exceptions=True)
            elapsed = time.perf_counter() - start
            # 验证所有连接稳定
            exceptions = [m for m in messages if isinstance(m, Exception)]
            assert len(exceptions) == 0, f"{len(exceptions)} 个连接异常"
            assert elapsed < 5.0, f"消息接收耗时 {elapsed:.3f}s"
        finally:
            for ws in clients:
                await ws.close()

    @pytest.mark.asyncio
    async def test_tc_perf_005_log_query_10000(
        self, client: AsyncClient, session
    ):
        """TC-PERF-005: 日志查询 10000 条性能测试"""
        # 预置 10000 条日志（通过模型工厂或直接插入）
        from app.models.log import LogEntry
        from datetime import datetime
        for i in range(10000):
            log = LogEntry(
                level="INFO",
                message=f"测试日志消息 {i}",
                timestamp=datetime.utcnow()
            )
            session.add(log)
        session.commit()

        start = time.perf_counter()
        response = await client.get("/api/logs?limit=200")
        elapsed = time.perf_counter() - start
        assert response.status_code == 200
        assert elapsed < 0.5, f"响应时间 {elapsed:.3f}s 超过 500ms 限制"

    @pytest.mark.asyncio
    async def test_tc_perf_006_concurrent_writes(
        self, client: AsyncClient, project
    ):
        """TC-PERF-006: 数据库并发写入测试"""
        async def create_character(i: int):
            response = await client.post(
                f"/api/projects/{project.id}/characters",
                json={"name": f"并发角色_{i}", "char_type": "extra"}
            )
            return response

        results = await asyncio.gather(*[
            create_character(i) for i in range(10)
        ])
        # 所有请求应成功，无死锁
        success_count = sum(
            1 for r in results if r.status_code == 200
        )
        assert success_count == 10, (
            f"仅 {success_count}/10 个并发写入成功"
        )
```
