```typescript
// tests/pages/HomePage.test.tsx
import { render, screen, waitFor } from '@testing-library/react';
import { describe, it, expect, vi } from 'vitest';
import { HomePage } from '@/pages/HomePage';
import { QueryClient, QueryClientProvider } from '@tanstack/react-query';
import { BrowserRouter } from 'react-router-dom';

const wrapper = ({ children }: { children: React.ReactNode }) => {
  const queryClient = new QueryClient({
    defaultOptions: { queries: { retry: false } },
  });
  return (
    <BrowserRouter>
      <QueryClientProvider client={queryClient}>
        {children}
      </QueryClientProvider>
    </BrowserRouter>
  );
};

describe('HomePage', () => {
  it('TC-FE-RENDER-001: 正确渲染项目列表', async () => {
    render(<HomePage />, { wrapper });

    await waitFor(() => {
      expect(screen.getByText('都市奇幻录')).toBeInTheDocument();
    });
  });

  it('TC-FE-RENDER-002: 空列表显示提示', async () => {
    // MSW handler 返回空列表
    // http.get('/api/projects', () => HttpResponse.json({ data: [] }))

    render(<HomePage />, { wrapper });

    await waitFor(() => {
      expect(screen.getByText(/暂无项目/i)).toBeInTheDocument();
    });
  });
});
```

```typescript
// tests/components/EntityEditDialog.test.tsx
import { render, screen, fireEvent, waitFor } from '@testing-library/react';
import userEvent from '@testing-library/user-event';
import { describe, it, expect, vi } from 'vitest';
import { EntityEditDialog } from '@/components/EntityEditDialog';
import { QueryClient, QueryClientProvider } from '@tanstack/react-query';
import { BrowserRouter } from 'react-router-dom';

const wrapper = ({ children }: { children: React.ReactNode }) => {
  const queryClient = new QueryClient({
    defaultOptions: { queries: { retry: false } },
  });
  return (
    <BrowserRouter>
      <QueryClientProvider client={queryClient}>
        {children}
      </QueryClientProvider>
    </BrowserRouter>
  );
};

describe('EntityEditDialog', () => {
  it('TC-FE-FORM-001: 正常创建项目', async () => {
    const onSubmit = vi.fn();
    render(
      <EntityEditDialog
        open={true}
        onClose={vi.fn()}
        entityType="project"
        onSubmit={onSubmit}
      />,
      { wrapper }
    );

    await userEvent.type(screen.getByLabelText(/项目名称/), '我的新项目');
    await userEvent.type(screen.getByLabelText(/描述/), '项目描述');
    await userEvent.click(screen.getByText('创建'));

    expect(onSubmit).toHaveBeenCalledWith(
      expect.objectContaining({ name: '我的新项目' })
    );
  });

  it('TC-FE-FORM-002: 必填验证', async () => {
    render(
      <EntityEditDialog
        open={true}
        onClose={vi.fn()}
        entityType="project"
        onSubmit={vi.fn()}
      />,
      { wrapper }
    );

    await userEvent.click(screen.getByText('创建'));

    expect(screen.getByText(/请输入项目名称/)).toBeInTheDocument();
  });
});
```

### 5.3 API 调用 Mock 测试

| 编号 | 名称 | 前置条件 | 步骤 | 预期结果 | 优先级 |
|------|------|----------|------|----------|--------|
| TC-FE-API-001 | 项目列表请求 | 配置 MSW handler | 组件挂载 | 正确发起 GET 请求 | P0 |

| TC-FE-API-003 | API 错误处理 | MSW 返回 500 | 组件挂载 | 显示错误 UI | P0 |

| TC-FE-API-007 | 缓存失效 | 已有缓存数据 | 更新操作后 | 缓存失效，重新获取 | P1 |

```typescript
// tests/api/projects.test.ts
import { http, HttpResponse } from 'msw';
import { server } from '@/mocks/server';
import { getProjects, createProject } from '@/api/projects';

describe('Projects API', () => {
  it('TC-FE-API-001: 获取项目列表', async () => {
    server.use(
      http.get('/api/projects', () => {
        return HttpResponse.json({
          data: [
            {
              id: 'a1b2c3d4-e5f6-7890-abcd-ef1234567890',
              name: '项目A',
              status: 'active',
            },
          ],
        });
      })
    );

    const result = await getProjects();
    expect(result).toHaveLength(1);
    expect(result[0].name).toBe('项目A');
  });

  it('TC-FE-API-003: API 错误处理', async () => {
    server.use(
      http.get('/api/projects', () => {
        return HttpResponse.json(
          { data: null, message: '服务器内部错误' },
          { status: 500 }
        );
      })
    );

    await expect(getProjects()).rejects.toThrow();
  });
});
```

```typescript
// tests/hooks/useTaskWebSocket.test.ts
import { renderHook, act } from '@testing-library/react';
import { useTaskWebSocket } from '@/hooks/useTaskWebSocket';

describe('useTaskWebSocket', () => {
  it('TC-FE-WS-001: 接收任务进度更新', () => {
    const { result } = renderHook(() => useTaskWebSocket());

    // 模拟 WebSocket 消息
    act(() => {
      const ws = result.current.wsRef.current;
      ws.onmessage?.({
        data: JSON.stringify({
          type: 'task_progress',
          data: { task_id: 1, progress: 50 },
        }),
      } as MessageEvent);
    });

    expect(result.current.progress).toBe(50);
  });
});
```

### 5.5 状态管理测试（Zustand）

| 编号 | 名称 | 前置条件 | 步骤 | 预期结果 | 优先级 |
|------|------|----------|------|----------|--------|
| TC-FE-STORE-001 | UI Store 初始化 | 无 | 创建 store | 初始状态正确（主题、Toast） | P0 |
| TC-FE-STORE-002 | 切换主题 | UI Store 已创建 | toggleTheme() | 主题状态切换 | P0 |
| TC-FE-STORE-003 | Toast 消息管理 | UI Store 已创建 | showToast() / dismissToast() | Toast 正确显示和消失 | P0 |
| TC-FE-STORE-004 | 日志 Store 初始化 | 无 | 创建 logStore | 初始状态正确（日志列表） | P1 |
| TC-FE-STORE-005 | 日志追加与清除 | logStore 已创建 | appendLog() / clearLogs() | 日志正确追加和清除 | P1 |

> **说明**: 业务数据（项目、角色、场景等）由 TanStack Query 管理，不通过 Zustand store。Zustand 仅用于 UI 状态（`stores/ui.ts`）和日志状态（`stores/logStore.ts`）。

```typescript
// tests/stores/ui.test.ts
import { useUIStore } from '@/stores/ui';

describe('useUIStore', () => {
  beforeEach(() => {
    useUIStore.setState({ theme: 'light', toasts: [] });
  });

  it('TC-FE-STORE-001: 初始状态', () => {
    const state = useUIStore.getState();
    expect(state.theme).toBe('light');
    expect(state.toasts).toEqual([]);
  });

  it('TC-FE-STORE-002: 切换主题', () => {
    useUIStore.getState().toggleTheme();
    expect(useUIStore.getState().theme).toBe('dark');
    useUIStore.getState().toggleTheme();
    expect(useUIStore.getState().theme).toBe('light');
  });

  it('TC-FE-STORE-003: Toast 消息管理', () => {
    useUIStore.getState().showToast({ message: '操作成功', type: 'success' });
    expect(useUIStore.getState().toasts).toHaveLength(1);
    expect(useUIStore.getState().toasts[0].message).toBe('操作成功');
  });
});

// tests/stores/logStore.test.ts
import { useLogStore } from '@/stores/logStore';

describe('useLogStore', () => {
  beforeEach(() => {
    useLogStore.setState({ logs: [] });
  });

  it('TC-FE-STORE-004: 初始状态', () => {
    const state = useLogStore.getState();
    expect(state.logs).toEqual([]);
  });

  it('TC-FE-STORE-005: 日志追加与清除', () => {
    useLogStore.getState().appendLog({
      level: 'info',
      message: '任务完成',
      timestamp: '2026-06-20T10:00:00+08:00',
    });
    expect(useLogStore.getState().logs).toHaveLength(1);

    useLogStore.getState().clearLogs();
    expect(useLogStore.getState().logs).toEqual([]);
  });
});
```
