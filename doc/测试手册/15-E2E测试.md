### 6.1 完整项目管理流程

| 编号 | 名称 | 前置条件 | 步骤 | 预期结果 | 优先级 |
|------|------|----------|------|----------|--------|
| TC-E2E-PROJ-001 | 项目完整生命周期 | 系统运行 | 1. 创建项目 2. 编辑项目 3. 查看项目详情 4. 归档项目 5. 删除项目 | 各步骤数据正确持久化 | P0 |
| TC-E2E-PROJ-002 | 多项目管理 | 系统运行 | 1. 创建3个项目 2. 列表查看 3. 切换项目 4. 删除其中一个 | 列表正确更新，切换无数据错乱 | P0 |

```typescript
// tests/e2e/project-lifecycle.spec.ts
import { test, expect } from '@playwright/test';

test.describe('项目完整生命周期', () => {
  test('TC-E2E-PROJ-001: 创建→编辑→归档→删除', async ({ page }) => {
    // 1. 导航到首页（项目列表）
    await page.goto('/');

    // 2. 创建项目
    await page.click('[data-testid="create-project-btn"]');
    await page.fill('[data-testid="project-name-input"]', 'E2E测试项目');
    await page.fill('[data-testid="project-desc-input"]', '自动化E2E测试');
    await page.click('[data-testid="submit-btn"]');

    // 3. 验证创建成功
    await expect(page.locator('[data-testid="entity-card"]'))
      .toContainText('E2E测试项目');

    // 4. 编辑项目
    await page.click('[data-testid="entity-card"]:first-child');
    await page.click('[data-testid="edit-btn"]');
    await page.fill('[data-testid="project-name-input"]', 'E2E已编辑');
    await page.click('[data-testid="save-btn"]');
    await expect(page.locator('h1')).toContainText('E2E已编辑');

    // 5. 归档项目
    await page.click('[data-testid="archive-btn"]');
    await expect(page.locator('[data-testid="status-badge"]'))
      .toContainText('archived');

    // 6. 删除项目
    await page.click('[data-testid="delete-btn"]');
    await page.click('[data-testid="confirm-delete-btn"]');
    await expect(page.locator('[data-testid="entity-card"]'))
      .toHaveCount(0);
  });
});
```

| TC-E2E-GEN-002 | 批量首帧生成 | 多个分镜有prompt | 1. 选择多集 2. 批量生成 3. 等待全部完成 | 所有分镜生成成功 | P0 |

| TC-E2E-GEN-005 | 取消生成 | 生成进行中 | 1. 开始生成 2. 点击取消 | 任务取消，状态更新 | P1 |
| TC-E2E-GEN-006 | 生成进度实时更新 | WebSocket已连接 | 1. 开始生成 2. 观察进度 | 进度条实时更新 | P0 |

```typescript
// tests/e2e/generation-flow.spec.ts
import { test, expect } from '@playwright/test';

test.describe('AI 生成流程', () => {
  test.beforeEach(async ({ page }) => {
    // 设置 Mock Provider
    await page.goto('/settings/providers');
    await page.click('[data-testid="add-provider"]');
    await page.fill('[data-testid="provider-name"]', 'Mock Provider');
    // ...
  });

  test('TC-E2E-GEN-001: 单张首帧生成', async ({ page }) => {
    // 导航到分镜页面
    await page.goto('/projects/1/episodes/1');

    // 点击第一个分镜的生成按钮
    await page.click('[data-testid="shot-1"] [data-testid="generate-first-frame"]');

    // 等待 WebSocket 进度消息
    await expect(page.locator('[data-testid="progress-bar"]'))
      .toBeVisible({ timeout: 30000 });

    // 验证生成完成
    await expect(page.locator('[data-testid="shot-1-image"]'))
      .toBeVisible({ timeout: 60000 });
  });

  test('TC-E2E-GEN-006: 进度实时更新', async ({ page }) => {
    await page.goto('/projects/1/episodes/1');
    await page.click('[data-testid="shot-1"] [data-testid="generate-first-frame"]');

    // 验证进度条动态更新
    const progressBar = page.locator('[data-testid="progress-bar"]');
    await expect(progressBar).toBeVisible();

    // 进度应该递增
    let prevProgress = 0;
    for (let i = 0; i < 5; i++) {
      const progress = await progressBar.getAttribute('aria-valuenow');
      expect(Number(progress)).toBeGreaterThanOrEqual(prevProgress);
      prevProgress = Number(progress);
      await page.waitForTimeout(1000);
    }
  });
});
```
