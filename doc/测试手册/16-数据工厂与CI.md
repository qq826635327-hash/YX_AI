# tests/factories.py
import factory
from datetime import datetime
from app.models import (
    Project, Character, Scene, Prop, Episode, Shot,
    GenerationTask, Asset, ApiProvider, WorkflowMapping
)

class ProjectFactory(factory.Factory):
    """项目工厂"""
    class Meta:
        model = Project

    name = factory.Sequence(lambda n: f"测试项目_{n}")
    description = factory.Faker("text", max_nb_chars=200)
    status = "active"
    root_path = factory.LazyAttribute(lambda obj: f"/tmp/projects/{obj.name}")

    @factory.lazy_attribute
    def created_at(self):
        return datetime.utcnow()

    @factory.lazy_attribute
    def updated_at(self):
        return datetime.utcnow()


class CharacterFactory(factory.Factory):
    """角色工厂"""
    class Meta:
        model = Character

    project_id = factory.SelfAttribute("project.id")
    name = factory.Sequence(lambda n: f"角色_{n}")
    char_type = factory.Iterator(["protagonist", "supporting", "extra"])
    description = factory.Faker("text", max_nb_chars=100)
    gen_status = "none"
    sort_order = factory.Sequence(lambda n: n)


class SceneFactory(factory.Factory):
    """场景工厂"""
    class Meta:
        model = Scene

    project_id = factory.SelfAttribute("project.id")
    name = factory.Sequence(lambda n: f"场景_{n}")
    description = factory.Faker("text", max_nb_chars=100)
    gen_status = "none"
    sort_order = factory.Sequence(lambda n: n)


class PropFactory(factory.Factory):
    """道具工厂"""
    class Meta:
        model = Prop

    project_id = factory.SelfAttribute("project.id")
    name = factory.Sequence(lambda n: f"道具_{n}")
    description = factory.Faker("text", max_nb_chars=100)
    gen_status = "none"
    sort_order = factory.Sequence(lambda n: n)


class EpisodeFactory(factory.Factory):
    """剧集工厂"""
    class Meta:
        model = Episode

    project_id = factory.SelfAttribute("project.id")
    episode_no = factory.Sequence(lambda n: n + 1)
    title = factory.LazyAttribute(lambda obj: f"第{obj.episode_no}集")
    summary = factory.Faker("text", max_nb_chars=200)
    sort_order = factory.Sequence(lambda n: n)


class ShotFactory(factory.Factory):
    """分镜工厂"""
    class Meta:
        model = Shot

    project_id = factory.SelfAttribute("project.id")
    episode_id = factory.SelfAttribute("episode.id")
    shot_no = factory.Sequence(lambda n: n + 1)
    summary = factory.Faker("text", max_nb_chars=100)
    first_frame_prompt = factory.Faker("text", max_nb_chars=200)
    sort_order = factory.Sequence(lambda n: n)


class GenerationTaskFactory(factory.Factory):
    """生成任务工厂"""
    class Meta:
        model = GenerationTask

    project_id = factory.SelfAttribute("project.id")
    target_type = factory.Iterator(["first_frame", "last_frame", "video"])
    target_id = factory.Faker("pyint")
    provider_type = "openai"
    status = factory.Iterator(["pending", "queued", "running", "succeeded", "failed"])
    progress = factory.LazyAttribute(
        lambda obj: 100 if obj.status == "succeeded" else 0
    )


class AssetFactory(factory.Factory):
    """素材工厂"""
    class Meta:
        model = Asset

    project_id = factory.SelfAttribute("project.id")
    asset_type = factory.Iterator(["image", "video"])
    category = "generated"
    file_path = factory.Faker("file_path", extension="png")
    file_name = factory.LazyAttribute(lambda obj: f"asset_{obj.id}.png")
    file_size = factory.Faker("pyint", min_value=1000, max_value=10000000)
    mime_type = factory.LazyAttribute(
        lambda obj: "image/png" if obj.asset_type == "image" else "video/mp4"

## 8. CI/CD 集成

### 8.1 GitHub Actions 配置

```yaml
# .github/workflows/test.yml
name: 自动化测试

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main, develop]

jobs:
  # ========== 后端测试 ==========
  backend-tests:
    runs-on: ubuntu-latest
    strategy:
      matrix:
        python-version: ["3.11", "3.12"]

    steps:
      - uses: actions/checkout@v4

      - name: 设置 Python ${{ matrix.python-version }}
        uses: actions/setup-python@v5
        with:
          python-version: ${{ matrix.python-version }}

      - name: 安装依赖
        run: |
          cd backend
          python -m pip install --upgrade pip
          pip install -r requirements.txt
          pip install pytest pytest-asyncio pytest-cov httpx factory-boy

      - name: 运行后端测试
        run: |
          cd backend
          pytest tests/ -v \
            --cov=app \
            --cov-report=xml \
            --cov-report=term-missing \
            --junitxml=reports/pytest-results.xml

      - name: 上传后端覆盖率报告
        uses: codecov/codecov-action@v4
        with:
          file: backend/coverage.xml
          flags: backend
          fail_ci_if_error: false

      - name: 上传测试报告
        uses: actions/upload-artifact@v4
        if: always()
        with:
          name: backend-test-report-${{ matrix.python-version }}
          path: backend/reports/

  # ========== 前端测试 ==========
  frontend-tests:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: 设置 Node.js
        uses: actions/setup-node@v4
        with:
          node-version: "20"
          cache: "npm"
          cache-dependency-path: frontend/package-lock.json

      - name: 安装依赖
        run: |
          cd frontend
          npm ci

      - name: 运行单元测试
        run: |
          cd frontend
          npx vitest run --coverage --reporter=json --outputFile=reports/vitest-results.json

      - name: 上传前端覆盖率报告
        uses: codecov/codecov-action@v4
        with:
          file: frontend/coverage/coverage-final.json
          flags: frontend
          fail_ci_if_error: false

  # ========== E2E 测试 ==========
  e2e-tests:
    runs-on: ubuntu-latest
    needs: [backend-tests, frontend-tests]
    steps:
      - uses: actions/checkout@v4

      - name: 设置 Python
        uses: actions/setup-python@v5
        with:
          python-version: "3.12"

      - name: 设置 Node.js
        uses: actions/setup-node@v4
        with:
          node-version: "20"

      - name: 安装 Playwright 浏览器
        run: |
          cd frontend
          npm ci
          npx playwright install --with-deps chromium

      - name: 启动后端服务
        run: |
          cd backend
          pip install -r requirements.txt
          python -m uvicorn app.main:app --host 0.0.0.0 --port 8000 &
          sleep 5  # 等待服务启动

      - name: 启动前端开发服务器
        run: |
          cd frontend
          npm run dev &
          sleep 5

      - name: 运行 E2E 测试
        run: |
          cd frontend
          npx playwright test

      - name: 上传 E2E 报告
        uses: actions/upload-artifact@v4
        if: always()
        with:
          name: playwright-report
          path: frontend/playwright-report/

  # ========== 质量门禁 ==========
  quality-gate:
    runs-on: ubuntu-latest
    needs: [backend-tests, frontend-tests, e2e-tests]
    steps:
      - name: 检查测试通过率
        run: |
          echo "所有测试已通过，质量门禁通过"
          # 可在此处添加额外的质量检查逻辑
```

### 8.5 测试数据库隔离策略

#### 8.5.1 SQLite 内存数据库

测试环境使用 SQLite 内存数据库（`sqlite+pysqlite:///:memory:`），确保每个测试会话拥有独立的数据库实例，避免测试间数据污染。

#### 8.5.2 事务回滚隔离

每个测试函数使用独立的数据库事务，测试完成后自动回滚，确保数据库状态在每个测试结束后恢复原样：

```python
# tests/conftest.py（事务回滚隔离策略）
import pytest
from sqlmodel import SQLModel, Session, create_engine
from sqlmodel.pool import StaticPool

TEST_DATABASE_URL = "sqlite+pysqlite:///:memory:"

@pytest.fixture(scope="session")
def engine():
    """创建测试数据库引擎（session 级别，整个测试会话共享）"""
    engine = create_engine(
        TEST_DATABASE_URL,
        connect_args={"check_same_thread": False},
        poolclass=StaticPool,
    )
    SQLModel.metadata.create_all(engine)
    return engine

@pytest.fixture(name="session")
def session_fixture(engine):
    """每个测试独立的数据库会话 + 事务回滚"""
    connection = engine.connect()
    transaction = connection.begin()
    session = Session(bind=connection, expire_on_commit=False)
    yield session
    session.close()
    transaction.rollback()
    connection.close()
```

#### 8.5.3 yield 模式确保清理

使用 `yield` 模式而非 `return`，确保每个 fixture 在测试完成后执行清理逻辑（关闭会话、回滚事务、关闭连接），即使测试抛出异常也能正确清理。

#### 8.5.4 文件系统 Mock

使用 pytest 内置的 `tmp_path` fixture 隔离文件系统操作，避免测试对真实项目目录的读写：

```python
@pytest.fixture
def project_root(tmp_path):
    """使用临时目录作为项目根目录"""
    root = tmp_path / "projects"
    root.mkdir()
    return root

def test_tc_unit_biz_007_update_rename_dir(
    session, project, character, tmp_path
):
    """TC-UNIT-BIZ-007: update_entity 同步重命名目录"""

        assert updated.name == "新角色名"
```

# tests/utils.py
import pytest
from httpx import AsyncClient
from typing import Any

async def create_project(
    client: AsyncClient,
    name: str = "测试项目",
    **kwargs
) -> dict:
    """快速创建测试项目"""
    payload = {"name": name, **kwargs}
    response = await client.post("/api/projects", json=payload)
    assert response.status_code == 200
    return response.json()

async def create_character(
    client: AsyncClient,
    project_id: int,
    name: str = "测试角色",
    char_type: str = "protagonist",
    **kwargs
) -> dict:
    """快速创建测试角色"""
    payload = {"name": name, "char_type": char_type, **kwargs}
    response = await client.post(
        f"/api/projects/{project_id}/characters",
        json=payload
    )
    assert response.status_code == 200
    return response.json()

async def create_episode(
    client: AsyncClient,
    project_id: int,
    episode_no: int = 1,
    **kwargs
) -> dict:
    """快速创建测试剧集"""
    payload = {
        "episode_no": episode_no,
        "title": f"第{episode_no}集",
        **kwargs
    }
    response = await client.post(
        f"/api/projects/{project_id}/episodes",
        json=payload
    )
    assert response.status_code == 200
    return response.json()

async def create_shot(
    client: AsyncClient,
    episode_id: int,
    shot_no: int = 1,
    **kwargs
) -> dict:
    """快速创建测试分镜"""
    payload = {"shot_no": shot_no, "summary": "测试分镜", **kwargs}
    response = await client.post(
        f"/api/episodes/{episode_id}/shots",
        json=payload
    )
    assert response.status_code == 200
    return response.json()

async def assert_status(
    response: Any,
    expected_status: int
) -> dict:
    """断言响应状态码并返回JSON"""
    assert response.status_code == expected_status, (
        f"Expected {expected_status}, got {response.status_code}: "
        f"{response.text}"
    )
    return response.json() if response.text else {}
